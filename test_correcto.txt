function generar_codigo_residuo(nivel_peligrosidad, cantidad_kilos, destino) {
    if (cantidad_kilos > 450) {
        return "Error";
    }

    codigo = upper(slice(nivel_peligrosidad,3)) + "-" + cantidad_kilos + "-" + upper(slice(destino,3));

    return codigo;
}

generar_codigo_residuo("Alto", 300, "Incineracion");
